# 1. 进入 AI 目录
cd /opt/StreamLake-Binance/ai-engine

# 2. 激活虚拟环境 (这一步非常重要！)
source .venv/bin/activate

# 3. 后台启动：绑定 0.0.0.0 端口 8000，日志输出到 ai.log
nohup uvicorn main:app --host 0.0.0.0 --port 8000 > ai.log 2>&1 &

# 4. 退出虚拟环境（如果还要操作别的）
deactivate

# 1. 进入 API 目录
cd /opt/StreamLake-Binance/api-server/target

# 2. 先强杀可能存在的旧进程，防止端口冲突
fuser -k 8080/tcp

# 3. 后台启动：指定生产/测试环境配置（如果有），日志输出到 api.log
nohup java -jar api-server-1.0.0-SNAPSHOT.jar > api.log 2>&1 &
